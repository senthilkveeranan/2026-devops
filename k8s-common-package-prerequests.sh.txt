#!/bin/bash

set -e

echo "========== Kubernetes Common Setup (All Nodes) =========="

### 1. Disable SELinux
echo "[INFO] Disabling SELinux..."
setenforce 0 || true
sed -i 's/^SELINUX=enforcing/SELINUX=permissive/' /etc/selinux/config

### 2. Disable Swap
echo "[INFO] Disabling Swap..."
swapoff -a
sed -i '/swap/d' /etc/fstab

### 3. Load Kernel Modules
echo "[INFO] Configuring kernel modules..."
cat <<EOF | tee /etc/modules-load.d/k8s.conf
overlay
br_netfilter
EOF

modprobe overlay
modprobe br_netfilter

### 4. Sysctl Settings
echo "[INFO] Applying sysctl settings..."
cat <<EOF | tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                 = 1
EOF

sysctl --system

### 5. Install Required Packages
echo "[INFO] Installing base packages..."
dnf install -y yum-utils device-mapper-persistent-data lvm2 iproute-tc conntrack

### 6. Install Container Runtime (containerd)
echo "[INFO] Installing containerd..."
dnf install -y containerd

mkdir -p /etc/containerd
containerd config default | tee /etc/containerd/config.toml

sed -i 's/SystemdCgroup = false/SystemdCgroup = true/' /etc/containerd/config.toml

systemctl enable --now containerd

### 7. Add Kubernetes Repository
echo "[INFO] Adding Kubernetes repo..."
cat <<EOF | tee /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=https://pkgs.k8s.io/core:/stable:/v1.29/rpm/
enabled=1
gpgcheck=1
gpgkey=https://pkgs.k8s.io/core:/stable:/v1.29/rpm/repodata/repomd.xml.key
EOF

### 8. Install Kubernetes Components
echo "[INFO] Installing kubelet, kubeadm, kubectl..."
dnf install -y kubelet kubeadm kubectl

systemctl enable kubelet

### 9. Configure Time Sync (Chrony)
echo "[INFO] Configuring time synchronization..."
dnf install -y chrony
systemctl enable --now chronyd
chronyc makestep

timedatectl set-timezone Asia/Kolkata

echo "========== Common Setup Completed Successfully =========="

echo ""
echo "👉 Next Steps:"
echo "Master Node: kubeadm init --apiserver-advertise-address=<MASTER-IP>"
echo "Worker Node: kubeadm join <MASTER-IP>:6443 --token <TOKEN> --discovery-token-ca-cert-hash sha256:<HASH>"
